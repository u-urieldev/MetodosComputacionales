Regex Patterns for JSON

Keys: 
"(\w+)\s*":

Puntuation:
\{|\}|:|,

Strings:
"(\s*[^"])*"

Numbers:
-?\d+\.?\d*([eE]?[-\+]?\d+)?

Null:
null

Boolean:
true|false


string:
"(\s*\w+[^"])*"(?!\s*:)|(?!)

-> "(\s*[^" [])*"(?!\s*:)|(?!)


key: "(\s*[^" [])*"(?=\s*:)|(?!)

string: "(\s*[^" [])*"(?!<)|(?!)
